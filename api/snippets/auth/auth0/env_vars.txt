# Auth0 -- required for both frontend and backend
# Create an application at https://manage.auth0.com

# Frontend -- React + Vite reads VITE_*, Next.js reads NEXT_PUBLIC_* (set whichever applies)
VITE_AUTH0_DOMAIN=your-tenant.auth0.com
VITE_AUTH0_CLIENT_ID=your_client_id
VITE_AUTH0_AUDIENCE=https://your-tenant.auth0.com/api/v2/
NEXT_PUBLIC_AUTH0_DOMAIN=your-tenant.auth0.com
NEXT_PUBLIC_AUTH0_CLIENT_ID=your_client_id
NEXT_PUBLIC_AUTH0_AUDIENCE=https://your-tenant.auth0.com/api/v2/

# Backend (.env in server/)
AUTH0_DOMAIN=your-tenant.auth0.com
AUTH0_AUDIENCE=https://your-tenant.auth0.com/api/v2/
# AUTH0_AUDIENCE must match the API identifier in Auth0 Dashboard -> APIs
